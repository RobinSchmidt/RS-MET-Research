\subsubsection{Integration by Parts} 
\begin{align}
  \int_a^b u(x) v'(x) \, dx 
  & = \Big[u(x) v(x)\Big]_a^b - \int_a^b u'(x) v(x) \, dx\\[6pt]
  & = u(b) v(b) - u(a) v(a) - \int_a^b u'(x) v(x) \, dx.
\end{align}
With $u = u(x)$, $du = u'(x) \,dx$ such that $v = v(x)$ and $dv = v'(x) dx$:
\begin{align}
  \int u \, dv \ =\ uv - \int v \, du
\end{align}
