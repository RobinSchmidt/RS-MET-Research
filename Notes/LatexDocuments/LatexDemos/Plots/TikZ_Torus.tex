\pgfplotsset{compat=1.12}

\begin{tikzpicture}
  \begin{axis}[view={60}{60}]
    \addplot3[surf, colormap/viridis, samples = 30, samples y = 30,
              domain=0:2*pi, y domain=0:2*pi, z buffer=sort]
    ({(3+cos(deg(x)))*cos(deg(y+pi/2))}, 
     {(3+cos(deg(x)))*sin(deg(y+pi/2))}, 
     {sin(deg(x))});
  \end{axis}
\end{tikzpicture}
