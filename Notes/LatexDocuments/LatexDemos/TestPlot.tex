\pgfplotsset{width=12cm,height=12cm,compat=1.12}
\begin{tikzpicture}[thick, >=stealth']
\begin{axis}[xlabel={$x$}, ylabel={$y$}, view={25}{40}, 
             xmin=-1.5, xmax=1.5, ymin=-1.5, ymax=+1.5, samples=11]
            
  \fill[black] (0,0,0) circle (4pt);
  \draw[->, very thick, rsRed]   (0,0,0) -- (1,0,0);
  \draw[->, very thick, rsGreen] (0,0,0) -- (0,1,0);
  \draw[->, very thick, rsBlue]  (0,0,0) -- (0,0,1);           
  \addplot3[surf, mesh, thick, color=black, draw opacity = 0.4, domain=-1.2:1.2, y domain = -1.2:1.2] 
  {x^2 - y^2};

\end{axis}
\end{tikzpicture}
